\HeaderA{numdim}{the number of dimensions in image}{numdim}
\keyword{manip}{numdim}
\begin{Description}\relax
generic function to access the number of dimensions (e.g. axes) in a
brain iamge
\end{Description}
\begin{Usage}
\begin{verbatim}
numdim(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object derived from "BrainSpace" or "BrainData" class 
\end{ldescription}
\end{Arguments}
\begin{Value}
an integer value for the number of dimensions
\end{Value}

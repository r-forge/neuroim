\HeaderA{dataFile}{access the name of the data file associated with a "BrainHeader" object}{dataFile}
\keyword{manip}{dataFile}
\begin{Description}\relax
acesses the name of the file stored in a "BrainHeader" instance.
\end{Description}
\begin{Usage}
\begin{verbatim}
dataFile(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] object deriving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
a character string returning the filename of the image data.
\end{Value}

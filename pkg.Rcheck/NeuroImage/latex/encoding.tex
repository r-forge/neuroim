\HeaderA{encoding}{retrieves the encoding method used for image data}{encoding}
\keyword{manip}{encoding}
\begin{Description}\relax
accessor for the way in which the image data is encoded on disk
(e.g. "raw", "gzip", etc.")
\end{Description}
\begin{Usage}
\begin{verbatim}
encoding(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object deriving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
a character string representing the encoding type
\end{Value}

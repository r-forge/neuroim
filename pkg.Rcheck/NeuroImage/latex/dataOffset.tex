\HeaderA{dataOffset}{gets the data offset in bytes for an image file}{dataOffset}
\keyword{manip}{dataOffset}
\begin{Description}\relax
function for accessing the offset in bytes marking the place at which
the image data is stored in the file.
\end{Description}
\begin{Usage}
\begin{verbatim}
dataOffset(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] obect derving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
an integer value for the byte offset
\end{Value}

\HeaderA{BrainVolume}{creates a new "BrainVolume" object}{BrainVolume}
\keyword{manip}{BrainVolume}
\begin{Description}\relax
function to create a new object of class "BrainVolume"
\end{Description}
\begin{Usage}
\begin{verbatim}
BrainVolume(data, space, indices = NULL)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{data}] a numeric vector containing image data 
\item[\code{space}] object of class \code{BrainSpace} describing the
geometry of the slice 
\item[\code{indices}] optional argument specifiying 1-d indices if data
vector is sparse 
\end{ldescription}
\end{Arguments}
\begin{Value}
an object of class \code{BrainVolume}
\end{Value}

\HeaderA{readHeader}{reads the header part of an image file}{readHeader}
\keyword{manip}{readHeader}
\begin{Description}\relax
a function to read an image header without loading associated image data
\end{Description}
\begin{Usage}
\begin{verbatim}
readHeader(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object derived from class "BrainFile" 
\end{ldescription}
\end{Arguments}
\begin{Value}
an object derived from class "BrainHeader"
\end{Value}

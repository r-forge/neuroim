\HeaderA{versionNum}{retrieves the version number associated with an image file format}{versionNum}
\keyword{manip}{versionNum}
\begin{Description}\relax
retrieves the version number associated with an image file format
\end{Description}
\begin{Usage}
\begin{verbatim}
versionNum(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object derived from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
a character string indicating the version number
\end{Value}

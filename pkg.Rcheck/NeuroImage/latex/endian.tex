\HeaderA{endian}{the endianess of the stored image data}{endian}
\keyword{manip}{endian}
\begin{Description}\relax
retrieves the endianess ("big" or "little") of the binary image data
\end{Description}
\begin{Usage}
\begin{verbatim}
endian(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an instance deriving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
character string which can be either "biug" or "little"
\end{Value}

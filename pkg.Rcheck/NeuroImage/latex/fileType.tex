\HeaderA{fileType}{the image file type}{fileType}
\keyword{manip}{fileType}
\begin{Description}\relax
the file format used to store the image data, e.g. Analyze7.5, NIFTI,
or AFNI
\end{Description}
\begin{Usage}
\begin{verbatim}
fileType(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object deriving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
character string of file format name
\end{Value}

\HeaderA{dataType}{a string representing the data type of the image data stored on
file}{dataType}
\keyword{manip}{dataType}
\begin{Description}\relax
accessor for the string representation for the image data as it is
stored on disk (which may be distinct from the sdtorage type in R)
\end{Description}
\begin{Usage}
\begin{verbatim}
dataType(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] object deriving from class "BrainHeader" 
\end{ldescription}
\end{Arguments}
\begin{Value}
character string representing the data type
\end{Value}

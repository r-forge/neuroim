\HeaderA{RegionSphere}{creation function for Spheric BrainRegion3D}{RegionSphere}
\begin{Description}\relax
creates a new BrainRegion3D instance with a spheric geometry.
\end{Description}
\begin{Usage}
\begin{verbatim}
        RegionSphere(bvol, centroid, radius)    
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{bvol}] the \code{BrainVolume} that on which the geometry of the region is based 
\item[\code{centroid}] the center coordinates (in grid space) of the region 
\item[\code{radius}] the radius of the sphere of the constructed region 
\end{ldescription}
\end{Arguments}

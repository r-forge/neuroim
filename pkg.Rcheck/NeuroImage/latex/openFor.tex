\HeaderA{openFor}{indicates whether the file is open for read or write}{openFor}
\keyword{manip}{openFor}
\begin{Description}\relax
function which indicates whether a "BrainFile" object is open for read
or write operations
\end{Description}
\begin{Usage}
\begin{verbatim}
openFor(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object derived from "BrainFile" class 
\end{ldescription}
\end{Arguments}
\begin{Value}
returns "r" for read and "w" for write
\end{Value}

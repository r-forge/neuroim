\HeaderA{index-class}{Virtual Class "index" - Simple Class for image indices}{index.Rdash.class}
\keyword{classes}{index-class}
\begin{Description}\relax
The class \code{"index"} is a virtual class used for
indices (in signatures) for image data indexing and sub-assignment of
brain image data.
\end{Description}
\begin{Section}{Objects from the Class}
A virtual Class: No objects may be created from it.
\end{Section}
\begin{Examples}
\begin{ExampleCode}
showClass("index")
\end{ExampleCode}
\end{Examples}

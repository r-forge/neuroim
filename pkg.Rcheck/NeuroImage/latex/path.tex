\HeaderA{path}{returns the full path of a "BrainFile" source}{path}
\keyword{manip}{path}
\begin{Description}\relax
returns the full path of a "BrainFile" source
\end{Description}
\begin{Usage}
\begin{verbatim}
path(x)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] an object derived from the "BrainFile" class  
\end{ldescription}
\end{Arguments}
\begin{Value}
character string representing the full path of the data file
\end{Value}
